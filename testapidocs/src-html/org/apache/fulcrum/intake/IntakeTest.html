108    @Test
109    public void testInterfaceMapTo() throws Exception
110    {
111        IntakeService is = (IntakeService) this.lookup( IntakeService.ROLE );
112        Group group = is.getGroup("LoginIfcGroup");
113        assertNotNull(group);
114
115        Field<?> userNameField = group.get("Username");
116
117        ParserService ps = (ParserService) this.lookup( ParserService.ROLE );
118        ValueParser pp = ps.getParser(DefaultParameterParser.class);
119
120        pp.setString(userNameField.getKey(), "Joe");
121        userNameField.init(pp);
122        userNameField.validate();
123
124        LoginForm form = new LoginForm();
125        group.setProperties(form);
126
127        assertEquals("Joe", form.getUsername(), "User names should be equal");
128    }

202    /**
203     * @throws Exception generic exception
204     */
205    @Test
206    public void testMultiValueField() throws Exception
207    {
208        IntakeService is = (IntakeService) this.lookup( IntakeService.ROLE );
209        Group group = is.getGroup("NumberTest");
210        assertNotNull(group);
211        Field<?> multiValueField = group.get("MultiIntegerTestField");
212        assertTrue( (multiValueField.getValidator() instanceof IntegerValidator), "The Default Validator of an intake Field type int should be IntegerValidator");
213        assertTrue( multiValueField.isMultiValued(), "An intake Field type int, which is multiValued, should be multiValued");
214
215        ParserService ps = (ParserService) this.lookup( ParserService.ROLE );
216        ValueParser pp = ps.getParser(DefaultParameterParser.class);
217
218        int[] values = new int[] { 1, 2 };
219        pp.add("nt_0mitf", values[0]);
220        pp.add("nt_0mitf", values[1]);
221        group.init(pp);
222
223        assertTrue( multiValueField.isSet(), "The field should be set");
224        assertTrue( multiValueField.isValidated(), "The field should be validated");
225        assertTrue( multiValueField.isValid(), "The field should be valid");
226        assertTrue( Arrays.equals(values, (int[])multiValueField.getValue()), "The field should have the value [1, 2]");
227    }
