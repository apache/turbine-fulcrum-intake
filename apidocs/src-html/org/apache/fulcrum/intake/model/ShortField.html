022import org.apache.fulcrum.intake.IntakeException;
023import org.apache.fulcrum.intake.validator.ShortValidator;
024
025/**
026 * Processor for short fields.
027 *
028 * @version $Id$
029 */
030public class ShortField
031        extends Field<Short>
032{
033    /** Serial version */
034        private static final long serialVersionUID = -5838127207028024370L;
035
036        /**
037     * Constructor.
038     *
039     * @param field xml field definition object
040     * @param group xml group definition object
041     * @throws IntakeException thrown by superclass
042     */
043    public ShortField(XmlField field, Group group)
044            throws IntakeException
045    {
046        super(field, group);
047    }
048
049    /**
050     * Sets the default value for an Short Field
051     *
052     * @param prop Parameter for the default values
053     */
054    @Override
055        public void setDefaultValue(String prop)
056    {
057        defaultValue = null;
058
059        if (prop == null)
060        {
061            return;
062        }
063
064        defaultValue = Short.valueOf(prop);
065    }
066
067    /**
068     * Set the empty Value. This value is used if Intake
069     * maps a field to a parameter returned by the user and
070     * the corresponding field is either empty (empty string)
071     * or non-existant.
072     *
073     * @param prop The value to use if the field is empty.
074     */
075    @Override
076        public void setEmptyValue(String prop)
077    {
078        emptyValue = null;
079
080        if (prop == null)
081        {
082            return;
083        }
084
085        emptyValue = Short.valueOf(prop);
086    }
087
088    /**
089     * Provides access to emptyValue such that the value returned will be
090     * acceptable as an argument parameter to Method.invoke.  Subclasses
091     * that deal with primitive types should ensure that they return an
092     * appropriate value wrapped in the object wrapper class for the
093     * primitive type.
094     *
095     * @return the value to use when the field is empty or an Object that
096     * wraps the empty value for primitive types.
097     */
098    @Override
099        protected Object getSafeEmptyValue()
100    {
101        if (isMultiValued())
102        {
103            return new short[0];
104        }
105        else
106        {
107            return (null == getEmptyValue())
108                    ? Short.valueOf((short) 0) : getEmptyValue();
109        }
110    }
111
112    /**
113     * A suitable validator.
114     *
115     * @return A suitable validator
116     */
117    @Override
118        protected String getDefaultValidator()
119    {
120        return ShortValidator.class.getName();
121    }
122
123    /**
124     * Sets the value of the field from data in the parser.
125     */
126    @Override
127        protected void doSetValue()
128    {
129        if (isMultiValued())
130        {
131            Integer[] inputs = parser.getIntObjects(getKey());
132            short[] values = new short[inputs.length];
133
134            for (int i = 0; i < inputs.length; i++)
135            {
136                values[i] = inputs[i] == null
137                        ? getEmptyValue().shortValue()
138                        : inputs[i].shortValue();
139            }
140
141            setTestValue(values);
142        }
143        else
144        {
145            Integer value = parser.getIntObject(getKey());
146
147            if (value == null)
148            {
149                setTestValue(getEmptyValue());
150            }
151            else
152            {
153                setTestValue(Short.valueOf(value.shortValue()));
154            }
155        }
156    }
157}
