022import java.text.DateFormat;
023import java.text.ParseException;
024import java.util.Date;
025
026import org.apache.commons.lang3.StringUtils;
027import org.apache.fulcrum.intake.IntakeException;
028import org.apache.fulcrum.intake.IntakeRuntimeException;
029import org.apache.fulcrum.intake.validator.DateStringValidator;
030import org.apache.fulcrum.intake.validator.Validator;
031
032
033/**
034 * Field for date inputs as free form text.  The parsing of date strings
035 * is dependent on any rules that are defined, so this field will expect that
036 * any validator will be (or extend) DateStringValidator.
037 *
038 * @author <a href="mailto:jmcnally@collab.net">John McNally</a>
039 * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
040 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
041 * @version $Id$
042 */
043public class DateStringField
044        extends Field<Date>
045{
046    /** Serial version */
047        private static final long serialVersionUID = 3659498567870484883L;
048
049        /** date format */
050    private DateFormat df = null;
051
052    /**
053     * Constructor.
054     *
055     * @param field xml field definition object
056     * @param group xml group definition object
057     * @throws IntakeException thrown by superclass
058     */
059    public DateStringField(XmlField field, Group group)
060            throws IntakeException
061    {
062        super(field, group);
063
064        Validator<?> validator = getValidator();
065        if (validator == null || !(validator instanceof DateStringValidator))
066        {
067            df = DateFormat.getInstance();
068            df.setLenient(true);
069        }
070    }
071
072    /**
073     * Sets the default value for a DateString field
074     *
075     * @param prop Parameter for the default values
076     */
077    @Override
078        public void setDefaultValue(String prop)
079    {
080        defaultValue = null;
081
082        if (prop == null)
083        {
084            return;
085        }
086
087        try
088        {
089            defaultValue = getDate(prop);
090        }
091        catch (ParseException e)
092        {
093            throw new IntakeRuntimeException("Could not parse " + prop
094                    + " into a valid Date for the default value", e);
095        }
096    }
097
098    /**
099     * Set the empty Value. This value is used if Intake
100     * maps a field to a parameter returned by the user and
101     * the corresponding field is either empty (empty string)
102     * or non-existant.
103     *
104     * @param prop The value to use if the field is empty.
105     */
106    @Override
107        public void setEmptyValue(String prop)
108    {
109        emptyValue = null;
110
111        if (prop == null)
112        {
113            return;
114        }
115
116        try
117        {
118            emptyValue = getDate(prop);
119        }
120        catch (ParseException e)
121        {
122            throw new IntakeRuntimeException("Could not parse " + prop
123                    + " into a valid Date for the empty value", e);
124        }
125    }
126
127    /**
128     * A suitable validator.
129     *
130     * @return "DateStringValidator"
131     */
132    @Override
133        protected String getDefaultValidator()
134    {
135        return DateStringValidator.class.getName();
136    }
137
138    /**
139     * Sets the value of the field from data in the parser.
140     */
141    @Override
142        protected void doSetValue()
143    {
144        if (isMultiValued())
145        {
146            String[] inputs = parser.getStrings(getKey());
147            Date[] values = new Date[inputs.length];
148            for (int i = 0; i < inputs.length; i++)
149            {
150                try
151                {
152                    values[i] = StringUtils.isNotEmpty(inputs[i])
153                            ? getDate(inputs[i]) : getEmptyValue();
154                }
155                catch (ParseException e)
156                {
157                    values[i] = null;
158                }
159            }
160            setTestValue(values);
161        }
162        else
163        {
164            String val = parser.getString(getKey());
165            try
166            {
167                setTestValue(StringUtils.isNotEmpty(val) ? getDate(val) : getEmptyValue());
168            }
169            catch (ParseException e)
170            {
171                setTestValue(null);
172            }
173        }
174    }
175
176    /**
177     * Parses a test date string using the Validator if it exists and
178     * is an instance of DateStringValidator.  Otherwise, DateFormat.parse()
179     * is used.
180     *
181     * @param dateString The string date to parse
182     * @return A <code>Date</code> object
183     * @throws ParseException The date could not be parsed.
184     */
185    private Date getDate(String dateString)
186            throws ParseException
187    {
188        Date date = null;
189        // FIXME: Canonicalize user-entered date strings.
190        Validator<?> validator = getValidator();
191        if (validator != null && validator instanceof DateStringValidator)
192        {
193            date = ((DateStringValidator) validator).parse(dateString);
194        }
195        else
196        {
197            date = df.parse(dateString);
198        }
199        return date;
200    }
201
202    /**
203     * returns a String representation
204     *
205     * @return a String representation
206     */
207    @Override
208        public String toString()
209    {
210        String s = null;
211        Object value = getValue();
212        Validator<?> validator = getValidator();
213        if (value == null)
214        {
215            s = "";
216        }
217        else if (value instanceof String)
218        {
219            s = (String) value;
220        }
221        else if (validator != null && validator instanceof DateStringValidator)
222        {
223            s = ((DateStringValidator) validator).format((Date) value);
224        }
225        else
226        {
227            s = df.format((Date) value);
228        }
229        return s;
230    }
231}
