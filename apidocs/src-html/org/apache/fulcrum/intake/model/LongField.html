022import org.apache.fulcrum.intake.IntakeException;
023import org.apache.fulcrum.intake.validator.LongValidator;
024
025/**
026 * Processor for long fields.
027 *
028 * @version $Id$
029 */
030public class LongField
031        extends Field<Long>
032{
033    /** Serial version */
034        private static final long serialVersionUID = -7142823708600937188L;
035
036        /**
037     * Constructor.
038     *
039     * @param field xml field definition object
040     * @param group xml group definition object
041     * @throws IntakeException thrown by superclass
042     */
043    public LongField(XmlField field, Group group)
044            throws IntakeException
045    {
046        super(field, group);
047    }
048
049    /**
050     * Sets the default value for an Long Field
051     *
052     * @param prop Parameter for the default values
053     */
054    @Override
055        public void setDefaultValue(String prop)
056    {
057        defaultValue = null;
058
059        if (prop == null)
060        {
061            return;
062        }
063
064        defaultValue = Long.valueOf(prop);
065    }
066
067    /**
068     * Set the empty Value. This value is used if Intake
069     * maps a field to a parameter returned by the user and
070     * the corresponding field is either empty (empty string)
071     * or non-existant.
072     *
073     * @param prop The value to use if the field is empty.
074     */
075    @Override
076        public void setEmptyValue(String prop)
077    {
078        emptyValue = null;
079
080        if (prop == null)
081        {
082            return;
083        }
084
085        emptyValue = Long.valueOf(prop);
086    }
087
088    /**
089     * Provides access to emptyValue such that the value returned will be
090     * acceptable as an argument parameter to Method.invoke.  Subclasses
091     * that deal with primitive types should ensure that they return an
092     * appropriate value wrapped in the object wrapper class for the
093     * primitive type.
094     *
095     * @return the value to use when the field is empty or an Object that
096     * wraps the empty value for primitive types.
097     */
098    @Override
099        protected Object getSafeEmptyValue()
100    {
101        if (isMultiValued())
102        {
103            return new long[0];
104        }
105        else
106        {
107            return (null == getEmptyValue()) ? Long.valueOf(0l) : getEmptyValue();
108        }
109    }
110
111    /**
112     * A suitable validator.
113     *
114     * @return A suitable validator
115     */
116    @Override
117        protected String getDefaultValidator()
118    {
119        return LongValidator.class.getName();
120    }
121
122    /**
123     * Sets the value of the field from data in the parser.
124     */
125    @Override
126        protected void doSetValue()
127    {
128        if (isMultiValued())
129        {
130            Long[] inputs = parser.getLongObjects(getKey());
131            long[] values = new long[inputs.length];
132
133            for (int i = 0; i < inputs.length; i++)
134            {
135                values[i] = inputs[i] == null
136                        ? getEmptyValue().longValue()
137                        : inputs[i].longValue();
138            }
139
140            setTestValue(values);
141        }
142        else
143        {
144            setTestValue(parser.getLongObject(getKey(), getEmptyValue()));
145        }
146    }
147}
