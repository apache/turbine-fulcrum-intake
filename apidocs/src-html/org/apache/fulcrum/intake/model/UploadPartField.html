022import javax.servlet.http.Part;
023
024import org.apache.fulcrum.intake.IntakeException;
025import org.apache.fulcrum.intake.IntakeRuntimeException;
026import org.apache.fulcrum.intake.validator.FileValidator;
027import org.apache.fulcrum.intake.validator.ValidationException;
028import org.apache.fulcrum.parser.ParameterParser;
029import org.apache.fulcrum.parser.ValueParser;
030
031/**
032 * @author <a href="mailto:jmcnally@collab.net">John McNally</a>
033 * @author <a href="mailto:hps@intermeta.de">Henning P. Schmiedehausen</a>
034 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
035 * @version $Id$
036 */
037public class UploadPartField
038        extends Field<Part>
039{
040    /** Serial version */
041        private static final long serialVersionUID = -963692413506822188L;
042
043        /**
044     * Constructor.
045     *
046     * @param field xml field definition object
047     * @param group xml group definition object
048     * @throws IntakeException thrown by superclass
049     */
050    public UploadPartField(XmlField field, Group group)
051            throws IntakeException
052    {
053        super(field, group);
054    }
055
056    /**
057     * It is not possible to set the default value for this field type.
058     * Calling this method with a non-null parameter will result in a
059     * IntakeRuntimeException
060     *
061     * @param prop Parameter for the default values
062     * @throws IntakeRuntimeException if the parameter is not null
063     */
064    @Override
065        public void setDefaultValue(String prop)
066    {
067        if (prop != null)
068        {
069            throw new IntakeRuntimeException(
070                    "Default values are not valid for "
071                    + this.getClass().getName());
072        }
073
074        defaultValue = null;
075    }
076
077    /**
078     * It is not possible to set the empty value for this field type.
079     * Calling this method with a non-null parameter will result in a
080     * IntakeRuntimeException
081     *
082     * @param prop Parameter for the empty values
083     * @throws IntakeRuntimeException if the parameter is not null
084     */
085    @Override
086        public void setEmptyValue(String prop)
087    {
088        if (prop != null)
089        {
090            throw new IntakeRuntimeException(
091                    "Empty values are not valid for "
092                    + this.getClass().getName());
093        }
094
095        emptyValue = null;
096    }
097
098    /**
099     * A suitable validator.
100     *
101     * @return A suitable validator
102     */
103    @Override
104        protected String getDefaultValidator()
105    {
106        return FileValidator.class.getName();
107    }
108
109    /**
110     * Method called when this field (the group it belongs to) is
111     * pulled from the pool.  The request data is searched to determine
112     * if a value has been supplied for this field.  if so, the value
113     * is validated.
114     *
115     * @param vp a <code>ValueParser</code> value
116     * @return a <code>Field</code> value
117     * @throws IntakeException if an error occurs
118     */
119    @Override
120        public Field<Part> init(ValueParser vp)
121            throws IntakeException
122    {
123        if (!(vp instanceof ParameterParser))
124        {
125            throw new IntakeException(
126                    "UploadPartFields can only be used with ParameterParser");
127        }
128
129        super.init(vp);
130
131        if (parser.containsKey(getKey()))
132        {
133            setSet(true);
134            validate();
135        }
136
137        return this;
138    }
139
140    /**
141     * Compares request data with constraints and sets the valid flag.
142     *
143     * @return the valid flag
144     */
145    @Override
146        public boolean validate()
147    {
148        ParameterParser pp = (ParameterParser) super.parser;
149        if (isMultiValued())
150        {
151            Part[] ss = pp.getParts(getKey());
152            // this definition of not set might need refined.  But
153            // not sure the situation will arise.
154            if (ss.length == 0)
155            {
156                setSet(false);
157            }
158
159            if (getValidator() != null)
160            {
161                for (int i = 0; i < ss.length; i++)
162                {
163                    try
164                    {
165                        ((FileValidator) getValidator()).assertValidity(ss[i]);
166                    }
167                    catch (ValidationException ve)
168                    {
169                        setMessage(ve.getMessage());
170                    }
171                }
172            }
173
174            if (isSet() && isValid())
175            {
176                doSetValue();
177            }
178        }
179        else
180        {
181            Part s = pp.getPart(getKey());
182            if (s == null || s.getSize() == 0)
183            {
184                setSet(false);
185            }
186
187            if (getValidator() != null)
188            {
189                try
190                {
191                    ((FileValidator) getValidator()).assertValidity(s);
192
193                    if (isSet())
194                    {
195                        doSetValue();
196                    }
197                }
198                catch (ValidationException ve)
199                {
200                    setMessage(ve.getMessage());
201                }
202            }
203            else if (isSet())
204            {
205                doSetValue();
206            }
207        }
208
209        return isValid();
210    }
211
212    /**
213     * Sets the value of the field from data in the parser.
214     */
215    @Override
216        protected void doSetValue()
217    {
218        ParameterParser pp = (ParameterParser) super.parser;
219        if (isMultiValued())
220        {
221            setTestValue(pp.getParts(getKey()));
222        }
223        else
224        {
225            setTestValue(pp.getPart(getKey()));
226        }
227    }
228}
