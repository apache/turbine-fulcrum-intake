022import javax.xml.bind.annotation.adapters.XmlAdapter;
023
024import org.apache.fulcrum.intake.IntakeException;
025
026/**
027 * Creates Field objects.
028 *
029 * @author <a href="mailto:tv@apache.org">Thomas Vandahl</a>
030 * @version $Id$
031 */
032public class FieldAdapter extends XmlAdapter<XmlField, Field<?>>
033{
034    /**
035     * Creates a Field object appropriate for the type specified
036     * in the xml file.
037     *
038     * @see javax.xml.bind.annotation.adapters.XmlAdapter#unmarshal(java.lang.Object)
039     */
040    @Override
041    public Field<?> unmarshal(XmlField xmlField) throws Exception
042    {
043        Field<?> field = null;
044        FieldType type = xmlField.getType();
045
046        if (type == null)
047        {
048            throw new IntakeException("An unsupported type has been specified for " +
049                    xmlField.getName() + " in group " + xmlField.getGroup().getIntakeGroupName());
050        }
051        else
052        {
053            field = type.getInstance(xmlField, xmlField.getGroup());
054        }
055
056        return field;
057    }
058
059    /**
060     * @see javax.xml.bind.annotation.adapters.XmlAdapter#marshal(java.lang.Object)
061     */
062    @Override
063    public XmlField marshal(Field<?> field) throws Exception
064    {
065        // This is never used in this context
066        return new XmlField();
067    }
068}
