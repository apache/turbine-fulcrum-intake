022import java.io.IOException;
023import java.io.InputStream;
024import java.io.UnsupportedEncodingException;
025import java.nio.charset.Charset;
026import java.util.regex.Matcher;
027import java.util.regex.Pattern;
028
029import javax.servlet.http.Part;
030
031/**
032 * A validator that will compare a Part testValue against the following
033 * constraints in addition to those listed in DefaultValidator.
034 *
035 * This validator can serve as the base class for more specific validators
036 *
037 * @author <a href="mailto:jmcnally@collab.net">John McNally</a>
038 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
039 * @author <a href="mailto:Colin.Chalmers@maxware.nl">Colin Chalmers</a>
040 * @version $Id$
041 */
042public class FileValidator
043        extends DefaultValidator<Part>
044{
045    private final static Pattern charsetPattern = Pattern.compile(".+charset\\s*=\\s*(.+)");
046
047    /**
048     * Default constructor
049     */
050    public FileValidator()
051    {
052        super();
053    }
054
055    /**
056     * Determine whether a testValue meets the criteria specified
057     * in the constraints defined for this validator
058     *
059     * @param testValue a <code>Part</code> to be tested
060     * @throws ValidationException containing an error message if the
061     * testValue did not pass the validation tests.
062     */
063    public void assertValidity(Part testValue)
064            throws ValidationException
065    {
066        byte[] fileData = new byte[(int) testValue.getSize()];
067        String contentType = testValue.getContentType();
068        String charset = Charset.defaultCharset().name();
069
070        if (contentType.contains("charset"))
071        {
072            Matcher matcher = charsetPattern.matcher(contentType);
073            if (matcher.matches())
074            {
075                charset = matcher.group(1);
076            }
077        }
078
079        try (InputStream fis = testValue.getInputStream())
080        {
081            int byteSize = fis.read(fileData);
082            if ( fileData.length != byteSize )
083            {
084                throw new ValidationException("Byte length mismatch found");
085            }
086            
087        }
088        catch (IOException e)
089        {
090            fileData = null;
091        }
092
093        String content = null;
094        try
095        {
096                if ( fileData != null )
097                {
098                        content = new String(fileData, charset);
099                }
100        }
101        catch (UnsupportedEncodingException e)
102        {
103            throw new ValidationException("Invalid charset " + charset);
104        }
105
106        super.assertValidity(content);
107    }
108}
