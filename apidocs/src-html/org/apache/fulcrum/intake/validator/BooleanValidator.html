022import java.text.ParseException;
023
024import org.apache.commons.lang3.StringUtils;
025
026/**
027 * Validator for boolean field types.<br><br>
028 *
029 * Values are validated by attempting to match the value to
030 * a list of strings for true and false values.  The string
031 * values are compared without regard to case.<br>
032 *
033 * Valid values for Boolean.TRUE:
034 * <ul>
035 * <li>TRUE</li>
036 * <li>T</li>
037 * <li>YES</li>
038 * <li>Y</li>
039 * <li>1</li>
040 * <li>ON</li>
041 * </ul>
042 * Valid values for Boolean.FALSE:
043 * <ul>
044 * <li>FALSE</li>
045 * <li>F</li>
046 * <li>NO</li>
047 * <li>N</li>
048 * <li>0</li>
049 * <li>OFF</li>
050 * </ul>
051 *
052 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
053 * @author <a href="mailto:Colin.Chalmers@maxware.nl">Colin Chalmers</a>
054 * @author <a href="mailto:jh@byteaction.de">J&uuml;rgen Hoffmann</a>
055 * @version $Id$
056 */
057public class BooleanValidator
058        extends DefaultValidator<Boolean>
059{
060    /** String values which would evaluate to Boolean.TRUE */
061    private static String[] trueValues = {"TRUE","T","YES","Y","1","ON"};
062
063    /** String values which would evaluate to Boolean.FALSE */
064    private static String[] falseValues = {"FALSE","F","NO","N","0","OFF"};
065
066    /**
067     * Default Constructor
068     */
069    public BooleanValidator()
070    {
071        super();
072    }
073
074    /**
075     * Determine whether a testValue meets the criteria specified
076     * in the constraints defined for this validator
077     *
078     * @param testValue a <code>String</code> to be tested
079     * @throws ValidationException containing an error message if the
080     * testValue did not pass the validation tests.
081     */
082    @Override
083        public void assertValidity(String testValue)
084            throws ValidationException
085    {
086        super.assertValidity(testValue);
087
088        if (required || StringUtils.isNotEmpty(testValue))
089        {
090            try
091            {
092                parse(testValue);
093            }
094            catch (ParseException e)
095            {
096                throw new ValidationException(e.getMessage());
097            }
098        }
099    }
100
101    /**
102     * Parses a string value into a Boolean object.
103     *
104     * @param stringValue the value to parse
105     * @return a <code>Boolean</code> object
106     * @throws ParseException if the value cannot be parsed to a boolean
107     */
108    public Boolean parse(String stringValue)
109            throws ParseException
110    {
111        Boolean result = null;
112
113        for (int cnt = 0;
114             cnt < Math.max(trueValues.length, falseValues.length); cnt++)
115        {
116            // Short-cut evaluation or bust!
117            if (cnt < trueValues.length &&
118                    stringValue.equalsIgnoreCase(trueValues[cnt]))
119            {
120                result = Boolean.TRUE;
121                break;
122            }
123
124            if (cnt < falseValues.length &&
125                    stringValue.equalsIgnoreCase(falseValues[cnt]))
126            {
127                result = Boolean.FALSE;
128                break;
129            }
130        }
131
132        if (result == null)
133        {
134            throw new ParseException(stringValue +
135                    " could not be converted to a Boolean", 0);
136        }
137        return result;
138    }
139}
